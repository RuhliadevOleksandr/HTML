CREATE DATABASE ToolArchive DEFAULT CHARACTER SET UTF8;
USE ToolArchive;

CREATE TABLE Worker (
	WorkerID SMALLINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
 	FirstName VARCHAR(20) NOT NULL,
    LastName VARCHAR(20) NOT NULL
 	) ENGINE = INNODB;

CREATE TABLE `Group` (
 	GroupID SMALLINT UNSIGNED AUTO_INCREMENT PRIMARY KEY, 
 	Name VARCHAR(20) NOT NULL,
 	) ENGINE = INNODB;
 
CREATE TABLE `Storage` (
 	StorageID SMALLINT UNSIGNED AUTO_INCREMENT PRIMARY KEY, 
 	Name VARCHAR(20) NOT NULL,
 	Adress VARCHAR(50) NOT NULL
 	) ENGINE = INNODB;

CREATE TABLE Tool (
 	ToolID SMALLINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
	Name VARCHAR(20) NOT NULL,
 	Quality VARCHAR(5) NOT NULL,
 	Description TEXT,
 	GroupID SMALLINT UNSIGNED NOT NULL,
 	FOREIGN KEY (GroupID) REFERENCES `Group` (GroupID)
 	ON DELETE CASCADE ON UPDATE CASCADE,
 	WorkerID SMALLINT UNSIGNED NOT NULL,
 	FOREIGN KEY (WorkerID) REFERENCES Worker (WorkerID)
 	ON DELETE CASCADE ON UPDATE CASCADE,
 	StorageID SMALLINT UNSIGNED NOT NULL,
 	FOREIGN KEY (StorageID) REFERENCES `Storage` (StorageID)
 	ON DELETE CASCADE ON UPDATE CASCADE
 	) ENGINE = INNODB;
